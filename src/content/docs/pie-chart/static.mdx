## Usage

```tsx
import { EvilPieChart } from "@/components/evilcharts/charts/pie-chart";
```

```tsx
const data = [
  { browser: "chrome", visitors: 275 },
  { browser: "safari", visitors: 200 },
  { browser: "firefox", visitors: 187 },
];

const chartConfig = {
  chrome: {
    label: "Chrome",
    colors: { light: ["#3b82f6"], dark: ["#60a5fa"] },
  },
  safari: {
    label: "Safari",
    colors: { light: ["#10b981"], dark: ["#34d399"] },
  },
  firefox: {
    label: "Firefox",
    colors: { light: ["#f59e0b"], dark: ["#fbbf24"] },
  },
} satisfies ChartConfig;
```

### Interactive Selection

When `isClickable` is enabled, you can use the `onSelectionChange` callback to handle selection events:

```tsx
<EvilPieChart
  data={data}
  dataKey="visitors"
  nameKey="browser"
  chartConfig={chartConfig}
  isClickable={true}
  onSelectionChange={(selection) => {
    if (selection) {
      console.log("Selected:", selection.dataKey, "Value:", selection.value);
    } else {
      console.log("Deselected");
    }
  }}
/>
```

```tsx
<EvilPieChart  
  data={data}
  dataKey="visitors"
  nameKey="browser"
  chartConfig={chartConfig}
/>
```

## API Reference

<ApiReferenceWrapper>
  <ApiReference>
    <ApiHeader
      propName="data"
      isRequired 
    />
    <ApiContent>
      Data used to display the chart. An array of objects where each object represents a sector.
      
      type: `TData[]` where `TData extends Record<string, unknown>`
    </ApiContent>
  </ApiReference>
  
  <ApiReference>
    <ApiHeader
      propName="dataKey"
      isRequired
    />
    <ApiContent>
      The key from your data objects to use for sector values (typically numeric values that determine sector size).
      
      type: `keyof TData & string`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="nameKey"
      isRequired
    />
    <ApiContent>
      The key from your data objects to use for sector names (typically string values used for labels and legend).
      
      type: `keyof TData & string`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="chartConfig"
      isRequired
    />
    <ApiContent>
      Configuration object that defines the chart's sectors. Each key should match a value from your `nameKey` field, with corresponding colors.
      
      type: `ChartConfig`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="className"
    />
    <ApiContent>
      Additional CSS classes to apply to the chart container.
      
      type: `string`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="innerRadius"
    />
    <ApiContent>
      The inner radius of the pie. Set to a value greater than 0 to create a donut chart. Can be a number (pixels) or percentage string.
      
      type: `number | string`
      
      default: `0`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="outerRadius"
    />
    <ApiContent>
      The outer radius of the pie. Can be a number (pixels) or percentage string.
      
      type: `number | string`
      
      default: `"80%"`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="cornerRadius"
    />
    <ApiContent>
      The border radius for the corners of each sector in pixels.
      
      type: `number`
      
      default: `0`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="paddingAngle"
    />
    <ApiContent>
      The angle of padding between sectors in degrees.
      
      type: `number`
      
      default: `0`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="startAngle"
    />
    <ApiContent>
      The starting angle of the pie chart in degrees (0 is 3 o'clock, 90 is 12 o'clock).
      
      type: `number`
      
      default: `0`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="endAngle"
    />
    <ApiContent>
      The ending angle of the pie chart in degrees. Set to less than 360 for a partial pie.
      
      type: `number`
      
      default: `360`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="showLabels"
    />
    <ApiContent>
      Whether to display labels on each sector.
      
      type: `boolean`
      
      default: `false`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="labelKey"
    />
    <ApiContent>
      The key from your data objects to use for label text when `showLabels` is enabled. If not provided, uses the `dataKey` value.
      
      type: `keyof TData & string`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="labelListProps"

    <ApiContent>

  </ApiReference>

  <ApiReference>

  <ApiReference>

    <ApiHeader
      propName="isClickable"

      type: `boolean`
      
      default: `false`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="onSelectionChange"
    />
    <ApiContent>
      Callback function that is called when a sector is selected or deselected. Receives an object with `dataKey` (sector name) and `value` (sector value), or `null` when deselected.

      type: `(selection: { dataKey: string; value: number } | null) => void`
    </ApiContent>
  </ApiReference>
