## Usage

```tsx
import { EvilLineChart } from "@/components/evilcharts/charts/line-chart";
```

```tsx
<EvilLineChart  
  xDataKey=""
  curveType="monotone"
  strokeVariant="solid" 
  activeDotVariant="default"
  data={data} 
  chartConfig={chartConfig}
/>
```

### Interactive Selection

When `isClickable` is enabled, you can use the `onSelectionChange` callback to handle selection events:

```tsx
<EvilLineChart
  data={data}
  chartConfig={chartConfig}
  isClickable={true}
  onSelectionChange={(selectedDataKey) => {
    if (selectedDataKey) {
      console.log("Selected:", selectedDataKey);
    } else {
      console.log("Deselected");
    }
  }}
/>
```

## API Reference

<ApiReferenceWrapper>
  <ApiReference>
    <ApiHeader
      propName="data"
      isRequired

    <ApiContent>
      Data used to display the chart. An array of objects where each object represents a data point.

  </ApiReference>
