## Usage

```tsx
import { EvilRadialChart } from "@/components/evilcharts/charts/radial-chart";
```

```tsx
const data = [
  { browser: "chrome", visitors: 275 },
  { browser: "safari", visitors: 200 },
  { browser: "firefox", visitors: 187 },
];

const chartConfig = {
  chrome: {
    label: "Chrome",
    colors: { light: ["#3b82f6"], dark: ["#60a5fa"] },
  },
  safari: {
    label: "Safari",
    colors: { light: ["#10b981"], dark: ["#34d399"] },
  },
  firefox: {
    label: "Firefox",
    colors: { light: ["#f59e0b"], dark: ["#fbbf24"] },
  },
} satisfies ChartConfig;
```

### Interactive Selection

When `isClickable` is enabled, you can use the `onSelectionChange` callback to handle selection events:

```tsx
<EvilRadialChart
  data={data}
  dataKey="visitors"
  nameKey="browser"
  chartConfig={chartConfig}
  isClickable={true}
  onSelectionChange={(selection) => {
    if (selection) {
      console.log("Selected:", selection.dataKey, "Value:", selection.value);
    } else {
      console.log("Deselected");
    }
  }}
/>
```

```tsx
<EvilRadialChart  
  data={data}
  dataKey="visitors"
  nameKey="browser"
  chartConfig={chartConfig}
/>
```

## API Reference

<ApiReferenceWrapper>
  <ApiReference>
    <ApiHeader
      propName="data"
      isRequired 
    />
    <ApiContent>
      Data used to display the chart. An array of objects where each object represents a radial bar.
      
      type: `TData[]` where `TData extends Record<string, unknown>`
    </ApiContent>
  </ApiReference>
  
  <ApiReference>
    <ApiHeader
      propName="dataKey"
      isRequired
    />
    <ApiContent>
      The key from your data objects to use for bar values (typically numeric values that determine bar size).
      
      type: `keyof TData & string`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="nameKey"
      isRequired
    />
    <ApiContent>
      The key from your data objects to use for bar names (typically string values used for labels and legend).
      
      type: `keyof TData & string`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="chartConfig"
      isRequired
    />
    <ApiContent>
      Configuration object that defines the chart's bars. Each key should match a value from your `nameKey` field, with corresponding colors.
      
      type: `ChartConfig`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="className"
    />
    <ApiContent>
      Additional CSS classes to apply to the chart container.
      
      type: `string`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="variant"
    />
    <ApiContent>
      The visual variant of the radial chart. `"full"` displays a full circle (360°), `"semi"` displays a half circle (180°).
      
      type: `"full" | "semi"`
      
      default: `"full"`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="innerRadius"
    />
    <ApiContent>
      The inner radius of the radial bars. Can be a number (pixels) or percentage string.
      
      type: `number | string`
      
      default: `"30%"`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="outerRadius"
    />
    <ApiContent>
      The outer radius of the radial bars. Can be a number (pixels) or percentage string.
      
      type: `number | string`
      
      default: `"100%"`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="cornerRadius"
    />
    <ApiContent>
      The border radius for the corners of each bar in pixels.
      
      type: `number`
      
      default: `5`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="barSize"
    />
    <ApiContent>
      The thickness of each radial bar in pixels.
      
      type: `number`
      
      default: `14`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="hideTooltip"
    />
    <ApiContent>
      Whether to hide the tooltip on hover.
      
      type: `boolean`
      
      default: `false`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="tooltipRoundness"
    />
    <ApiContent>
      Controls the border-radius of the tooltip.
      
      type: `"sm" | "md" | "lg" | "xl"`
      
      default: `"lg"`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="tooltipVariant"
    />
    <ApiContent>
      Controls the visual style of the tooltip.
      
      type: `"default" | "frosted-glass"`
      
      default: `"default"`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="tooltipDefaultIndex"
    />
    <ApiContent>
      When set, the tooltip will be visible by default at the specified data point index.
      
      type: `number`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="legendVariant"
    />
    <ApiContent>
      The visual style variant for the legend indicators.
      
      type: `"square" | "circle" | "circle-outline" | "rounded-square" | "rounded-square-outline" | "vertical-bar" | "horizontal-bar"`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="hideLegend"
    />
    <ApiContent>
      Whether to hide the chart legend.
      
      type: `boolean`
      
      default: `false`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="backgroundVariant"
    />
    <ApiContent>
      Background pattern variant to display behind the chart.
      
      type: `BackgroundVariant`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="hideBackground"
    />
    <ApiContent>
      Whether to hide the background bars (the unfilled portion of each bar).
      
      type: `boolean`
      
      default: `false`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="isClickable"
    />
    <ApiContent>
      Enables interactive clicking on bars to select/deselect them. Selected bars expand slightly and unselected bars dim.
      
      type: `boolean`
      
      default: `false`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="onSelectionChange"
    />
    <ApiContent>
      Callback function that is called when a bar is selected or deselected. Receives an object with `dataKey` (bar name) and `value` (bar value), or `null` when deselected.
      
      **Note:** This prop is only available when `isClickable` is set to `true`.
      
      type: `(selection: { dataKey: string; value: number } | null) => void`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="isLoading"
    />
    <ApiContent>
      Shows a loading placeholder animation when data is being fetched.
      
      type: `boolean`
      
      default: `false`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="glowingBars"
    />
    <ApiContent>
      Array of bar names (values from your `nameKey` field) that should have a glowing effect applied. Creates a smooth outer glow around the specified bars.
      
      type: `string[]`
      
      default: `[]`
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="chartProps"
    />
    <ApiContent>
      Additional props to pass to the underlying Recharts RadialBarChart component.
      
      type: `ComponentProps<typeof RadialBarChart>`
      
      Read the <Link href="https://recharts.github.io/en-US/api/RadialBarChart/" _blank>Recharts RadialBarChart documentation</Link> for available props.
    </ApiContent>
  </ApiReference>

  <ApiReference>
    <ApiHeader
      propName="radialBarProps"
    />
    <ApiContent>
      Additional props to pass to the underlying Recharts RadialBar component.
      
      type: `Omit<ComponentProps<typeof RadialBar>, "dataKey">`
      
      Read the <Link href="https://recharts.github.io/en-US/api/RadialBar/" _blank>Recharts RadialBar documentation</Link> for available props.
    </ApiContent>
  </ApiReference>
</ApiReferenceWrapper>
